import json
import re

# Assume that your data is stored in a JSON file named "data.json"
with open("data.json", 'r') as f:
    data = json.load(f)

# Initialize an empty list to store all steps
steps = []

# Iterate over each item in the data
for item in data:
    # Get the response string
    response_str = item['response']['response']

    # Use a regular expression to find all substrings within squiggly brackets
    steps_in_item = re.findall(r'\{([^}]*)\}', response_str)

    # Add the steps from this item to the overall steps list
    steps.extend(steps_in_item)

print(steps)
